## QEMU 启动命令与内存压力施加

### 1. 无内存超售测试
- **目的**：测试在没有宿主机内存压力情况下的 QEMU 常规运行状态和性能。
- **配置**：仅启动主 VM。

### 2. 内存超售测试
- **目的**：测试在宿主机存在严重内存压力情况下的 QEMU 稳定性与表现。
### 1. 无超售方式启动命令
直接应用以下命令启动主 VM：

```bash
sudo env \
  VIO_DEBUG_TOUCH=0 \
  VIO_VDPA_THRESHOLD=1 \
  VIO_HIGH_IOPS=100000 \
  VIO_LOW_IOPS=20000 \
  VIO_LOW_WINDOWS=5 \
  VIO_PASSTHROUGH_MIN_NS=10000000000 \
  VIO_COOLDOWN_NS=3000000000 \
  VIO_VDPA_DEV=vdpa0 \
  /home/ss/qemu-VIO/build/qemu-system-aarch64 \
    -accel kvm \
    -machine virt,memory-backend=mem1 \
    -cpu host \
    -smp 4 \
    -m 4096 \
    -object memory-backend-ram,id=mem1,size=4096M,share=on,prealloc=yes \
    -bios /usr/share/edk2/aarch64/QEMU_EFI.fd \
    -drive if=virtio,format=qcow2,file=/home/ss/ubuntu-arm64/main-vm.qcow2 \
    -netdev vhost-vdpa,vhostdev=/dev/vhost-vdpa-0,id=n0,queues=4 \
    -device virtio-net-pci,netdev=n0,mac=52:54:00:cb:45:11,mq=on,vectors=10,iommu_platform=on,disable-legacy=on,page-per-vq=on \
    -nographic \
    -d guest_errors \
    -D /tmp/qemu-vio.log \
    2>&1 | tee /tmp/qemu-console.log
```

### 2. 内存超售方式启动命令
主 VM 的启动命令与**无超售方式**一致，但在启动主 VM 前，需要先执行以下命令开启 20 个占位 VM 施加内存压力：

```bash
./multi_vm.sh 20
```

#### 附：`multi_vm.sh` 脚本源码
```bash
#!/usr/bin/env bash
#
# 启动 N 台占位 VM，用于制造 host 内存压力
# 适配你的 aarch64 + KVM + hugepages 主 VM 环境
#
set -euo pipefail
###############################################################################
# 用法检查
###############################################################################
if [[ $# -lt 1 ]]; then
  echo "Usage: $0 <vm_count>" >&2
  exit 1
fi
VM_COUNT="$1"
if ! [[ "$VM_COUNT" =~ ^[0-9]+$ ]] || [[ "$VM_COUNT" -lt 1 ]]; then
  echo "vm_count must be a positive integer" >&2
  exit 1
fi
###############################################################################
# 参数配置（可通过环境变量覆盖）
###############################################################################
# 你自己环境的 QEMU 路径
QEMU_BIN="${QEMU_BIN:-/home/ss/vio/qemu-VIO/build/qemu-system-aarch64}"
# 基础镜像（和主 VM 共享同一个 base，只读）
BASE_IMG="${BASE_IMG:-/home/ss/ubuntu-arm64/ubuntu-arm64.qcow2}"
# 占位 VM 的 overlay 镜像存放目录
IMG_DIR="${IMG_DIR:-/home/ss/ubuntu-arm64/placeholders}"
# UEFI 固件
UEFI_BIN="${UEFI_BIN:-/usr/share/edk2/aarch64/QEMU_EFI.fd}"
# 占位 VM 绑定的 CPU 核（和主 VM 错开，防止干扰）
# 主 VM 如果用了 0-3，占位 VM 可以用 4-7 或更多
CPU_POOL="${CPU_POOL:-4,5,6,7}"
# 占位 VM 内存（MB）
# 建议按目标超分比设定
# 例：主机 32GB，主 VM 2GB，目标 1.5x 超分 → 需要额外消耗约 14GB
# 15台 × 1024MB = 15GB，加主 VM 2GB = 17GB，对 32GB 机器约 1.5x 超分
VM_MEM_MB="${VM_MEM_MB:-1024}"
# 占位 VM CPU 数（占位用，少给节省资源）
VM_SMP="${VM_SMP:-1}"
# SSH 转发端口起始（便于必要时登入排查）
SSH_BASE_PORT="${SSH_BASE_PORT:-2301}"
# QEMU monitor 端口起始（可用 telnet 连接发 quit 关机）
MON_BASE_PORT="${MON_BASE_PORT:-5601}"
# MAC 地址前缀
MAC_PREFIX="${MAC_PREFIX:-52:54:00:aa:bb}"
###############################################################################
# 前置检查
###############################################################################
if [[ ! -x "$QEMU_BIN" ]]; then
  echo "QEMU binary not found or not executable: $QEMU_BIN" >&2
  exit 1
fi
if [[ ! -f "$BASE_IMG" ]]; then
  echo "Base image not found: $BASE_IMG" >&2
  exit 1
fi
if [[ ! -f "$UEFI_BIN" ]]; then
  echo "UEFI firmware not found: $UEFI_BIN" >&2
  exit 1
fi
mkdir -p "$IMG_DIR"
###############################################################################
# 启动占位 VM
###############################################################################
for i in $(seq 1 "$VM_COUNT"); do
  idx=$((i - 1))
  overlay="${IMG_DIR}/placeholder${i}.qcow2"
  pidfile="${IMG_DIR}/placeholder${i}.pid"
  logfile="${IMG_DIR}/placeholder${i}.log"
  ssh_port=$((SSH_BASE_PORT + idx))
  mon_port=$((MON_BASE_PORT + idx))
  mac_last=$(printf "%02x" "$i")
  mac="${MAC_PREFIX}:${mac_last}"
  # 创建 overlay（如不存在）
  if [[ ! -f "$overlay" ]]; then
    echo "Creating overlay: $overlay"
    qemu-img create -f qcow2 -F qcow2 -b "$BASE_IMG" "$overlay" >/dev/null
  fi
  # 如果已在运行，跳过
  if [[ -f "$pidfile" ]] && kill -0 "$(cat "$pidfile")" 2>/dev/null; then
    echo "placeholder${i} already running (pid=$(cat "$pidfile")), skipping"
    continue
  fi
  echo "Starting placeholder${i}: mem=${VM_MEM_MB}MB  ssh=127.0.0.1:${ssh_port}  monitor=127.0.0.1:${mon_port}"
  # 注意：
  # 1. 占位 VM 用 memory-backend-ram（不用 hugepages，hugepages 留给主 VM）
  # 2. size="${VM_MEM_MB}M" 引号在 M 之后闭合，逗号在引号外
  # 3. 网络只用 user 模式，不用 vDPA（占位 VM 不需要高性能网络）
  # 4. -serial none -display none 替代 -nographic，避免 serial 输出干扰日志
  # 5. prealloc=yes 让内存在启动时立即占用物理页，真正制造内存压力
  nohup taskset -c "$CPU_POOL" sudo "$QEMU_BIN" \
    -accel kvm \
    -machine virt,memory-backend=mem1 \
    -cpu host \
    -smp "$VM_SMP" \
    -m "${VM_MEM_MB}" \
    -object memory-backend-ram,id=mem1,size="${VM_MEM_MB}M",share=on,prealloc=yes \
    -bios "$UEFI_BIN" \
    -drive if=virtio,format=qcow2,file="$overlay" \
    -netdev user,id="netadmin${i}",hostfwd=tcp::${ssh_port}-:22 \
    -device virtio-net-pci,netdev="netadmin${i}",mac="$mac",id="nic${i}" \
    -monitor "telnet:127.0.0.1:${mon_port},server,nowait" \
    -serial none \
    -display none \
    >"$logfile" 2>&1 &
  echo $! > "$pidfile"
  # 等待 2 秒确认进程存活
  sleep 2
  if ! kill -0 "$(cat "$pidfile")" 2>/dev/null; then
    echo "placeholder${i} failed to stay up, check $logfile" >&2
    cat "$logfile" >&2
    exit 1
  fi
  echo "placeholder${i} started (pid=$(cat "$pidfile"))"
done
###############################################################################
# 汇总输出
###############################################################################
echo ""
echo "=================================================="
echo "Started up to ${VM_COUNT} placeholder VM(s)"
echo "Each VM memory : ${VM_MEM_MB} MB"
echo "Total memory   : $(( VM_COUNT * VM_MEM_MB )) MB"
echo "Overlay dir    : $IMG_DIR"
echo "SSH ports      : ${SSH_BASE_PORT} ~ $((SSH_BASE_PORT + VM_COUNT - 1))"
echo "Monitor ports  : ${MON_BASE_PORT} ~ $((MON_BASE_PORT + VM_COUNT - 1))"
echo "=================================================="
echo ""
echo "Tips:"
echo "  Check memory pressure : free -h"
echo "  Stop one VM           : echo quit | telnet 127.0.0.1 <mon_port>"
echo "  Stop all placeholder  : $0 stop  (or kill pids in $IMG_DIR/*.pid)"

```

### 3. 关闭占位 VM（内存超售实验结束后）
内存超售实验结束后，需关闭所有占位 VM，推荐使用如下脚本：

```bash
cat > /home/ss/stop_placeholder.sh << 'EOF'
#!/usr/bin/env bash
# 停止所有占位 VM
IMG_DIR="${IMG_DIR:-/home/ss/ubuntu-arm64/placeholders}"
echo "=== Stopping all placeholder VMs ==="
for pidfile in "${IMG_DIR}"/*.pid; do
  [[ -f "$pidfile" ]] || continue
  pid=$(cat "$pidfile")
  name=$(basename "$pidfile" .pid)
  if kill -0 "$pid" 2>/dev/null; then
    echo "Stopping ${name} (pid=${pid})..."
    sudo kill "$pid"
    sleep 1
    # 确认是否已停止，若未停则强制 kill
    if kill -0 "$pid" 2>/dev/null; then
      echo "  Force killing ${name}..."
      sudo kill -9 "$pid"
    fi
  else
    echo "${name} already stopped"
  fi
  rm -f "$pidfile"
done
echo ""
echo "=== Result ==="
echo "Remaining QEMU processes:"
ps aux | grep qemu | grep -v grep || echo "None"
free -h
EOF
chmod +x /home/ss/stop_placeholder.sh
sudo bash /home/ss/stop_placeholder.sh
```

#### 1. 安装 Redis（host 与 guest 均需执行）
```bash

```

#### 3. host 测试连通性
```bash
redis-cli -h 192.168.101.20 ping
```

#### 4. host 运行性能测试脚本
将如下内容保存为 `redis_test.sh`，并赋予可执行权限：
```bash
#!/bin/bash

REDIS_IP="192.168.101.20"
REDIS_PORT=6379
REQUESTS=100000

CONCURRENCY_LIST=(1 5 10 20 50 100)
DATA_SIZES=(128 1470)

parse_csv_field() {
  local text="$1"
  local field="$2"
  echo "$text" | awk -F',' '
    NR > 1 && NF > 2 {
      gsub(/"/, "", $0)
      print $'"$field"'
      exit
    }'
}

echo "##############################################################"
echo "# Redis Benchmark Full Test"
echo "# Server: ${REDIS_IP}:${REDIS_PORT}"
echo "##############################################################"

for size in "${DATA_SIZES[@]}"; do
  echo
  echo "=============================================================="
  echo "  并发梯度测试"
  echo "  固定数据大小: ${size} Bytes | 总请求数: ${REQUESTS}"
  echo "=============================================================="
  printf "%-10s | %-12s | %-15s | %-15s\n" "并发连接" "命令" "吞吐(req/s)" "平均时延(ms)"
  echo "--------------------------------------------------------------"

  for c in "${CONCURRENCY_LIST[@]}"; do
    for cmd in set get; do
      OUTPUT=$(redis-benchmark -h $REDIS_IP -p $REDIS_PORT \
           -t $cmd -n $REQUESTS -c $c -d $size --csv 2>/dev/null)

      THROUGHPUT=$(parse_csv_field "$OUTPUT" 2)
      LATENCY=$(parse_csv_field "$OUTPUT" 3)

      if [ -z "$THROUGHPUT" ]; then
        THROUGHPUT="测试失败"
        LATENCY="-"
      fi

      printf "%-10s | %-12s | %-15s | %-15s\n" "$c" "$cmd" "$THROUGHPUT" "$LATENCY"
      sleep 2
    done
  done
done

echo "=============================================================="
```

执行：
```bash
chmod +x ./redis_test.sh
./redis_test.sh
```

#### 6. host 运行性能测试脚本
将如下内容保存为 `nginx_test.sh`，并赋予可执行权限：
```bash
#!/bin/bash

SERVER="http://192.168.101.20"
TEST_DIR="/test"
FILES=("1k.bin" "4k.bin" "64k.bin" "1m.bin")
CONCURRENCIES=(16 32 64 128 256)
REQUESTS=10000
THREADS=4
OUTPUT_FILE="httpress_report_$(date +%Y%m%d_%H%M%S).txt"

# 解析httpress输出
parse_httpress() {
  local output="$1"
  local total_requests="$2"

  local timing_line=$(echo "$output" | grep "TIMING:")
  local traffic_line=$(echo "$output" | grep "TRAFFIC:")

  # 提取RPS
  local rps=$(echo "$timing_line" | grep -oP '\\d+ rps' | grep -oP '\\d+')
  # 提取kbps
  local kbps=$(echo "$timing_line" | grep -oP '\\d+ kbps' | grep -oP '\\d+')
  # 提取avg req time (ms or us)
  local avg_time=$(echo "$timing_line" | grep -oP '[\\d.]+ ms avg' | grep -oP '[\\d.]+')
  local avg_time_unit="ms"

  if [ -z "$avg_time" ]; then
    avg_time=$(echo "$timing_line" | grep -oP '[\\d.]+ s avg' | grep -oP '[\\d.]+')
    avg_time_unit="s"
  fi

  # 转换带宽单位
  local mbps=$(echo "scale=2; $kbps / 1024" | bc)
  local transfer_sec=""
  if (( $(echo "$mbps >= 1024" | bc -l) )); then
    transfer_sec=$(echo "scale=2; $mbps / 1024" | bc)"GB"
  else
    transfer_sec="${mbps}MB"
  fi

  # 延迟单位处理
  local latency=""
  if [ "$avg_time_unit" = "ms" ]; then
    if (( $(echo "$avg_time < 1" | bc -l) )); then
      local us=$(echo "scale=2; $avg_time * 1000" | bc)
      latency="${us}us"
    else
      latency="${avg_time}ms"
    fi
  else
    latency="${avg_time}s"
  fi

  echo "$rps|$transfer_sec|$latency"
}

# 打印表头
print_header() {
  local file="$1"
  local mode="$2"
  echo ""
  echo "文件: $file [$mode]"
  printf "%-16s| %-16s| %-16s| %-16s\n" "目标并发" "Requests/sec" "Transfer/sec" "Avg Latency"
  echo "----------------------------------------------------------"
}

# 运行单次测试
run_test() {
  local url="$1"
  local concurrency="$2"
  local extra_args="$3"

  local output=$(httpress -n $REQUESTS -c $concurrency -t $THREADS $extra_args "$url" 2>/dev/null)
  echo "$output"
}

# 主测试函数
run_all_tests() {
  local log="$1"

  echo "==========================================================" | tee -a "$log"
  echo " HTTP 性能测试报告" | tee -a "$log"
  echo " 测试时间: $(date '+%Y-%m-%d %H:%M:%S')" | tee -a "$log"
  echo " 目标服务器: $SERVER" | tee -a "$log"
  echo "==========================================================" | tee -a "$log"

  for file in "${FILES[@]}"; do
    local url="${SERVER}${TEST_DIR}/${file}"

    # ======== HTTP 短连接 ========
    print_header "$file" "HTTP 短连接" | tee -a "$log"
    for c in "${CONCURRENCIES[@]}"; do
      local output=$(run_test "$url" "$c" "")
      local result=$(parse_httpress "$output" "$REQUESTS")
      local rps=$(echo $result | cut -d'|' -f1)
      local transfer=$(echo $result | cut -d'|' -f2)
      local latency=$(echo $result | cut -d'|' -f3)
      printf "%-16s| %-16s| %-16s| %-16s\n" "$c" "$rps" "$transfer" "$latency" | tee -a "$log"
    done

    # ======== HTTP 长连接 ========
    print_header "$file" "HTTP 长连接 (-k)" | tee -a "$log"
    for c in "${CONCURRENCIES[@]}"; do
      local output=$(run_test "$url" "$c" "-k")
      local result=$(parse_httpress "$output" "$REQUESTS")
      local rps=$(echo $result | cut -d'|' -f1)
      local transfer=$(echo $result | cut -d'|' -f2)
      local latency=$(echo $result | cut -d'|' -f3)
      printf "%-16s| %-16s| %-16s| %-16s\n" "$c" "$rps" "$transfer" "$latency" | tee -a "$log"
    done

    echo "==========================================================" | tee -a "$log"
  done
}

# 检查依赖
if ! command -v httpress &>/dev/null; then
  echo "[ERROR] httpress 未找到，请检查PATH或安装"
  exit 1
fi

if ! command -v bc &>/dev/null; then
  echo "[ERROR] bc 未找到，请安装: sudo apt install bc"
  exit 1
fi

# 验证文件可访问
echo "[INFO] 验证测试文件..."
for file in "${FILES[@]}"; do
  code=$(curl -s -o /dev/null -w "%{http_code}" "${SERVER}${TEST_DIR}/${file}")
  if [ "$code" != "200" ]; then
    echo "[WARN] ${file} 返回 HTTP $code，跳过"
  else
    echo "[OK] ${file} 可访问"
  fi
done
echo ""

# 开始测试
run_all_tests "$OUTPUT_FILE"

echo ""
echo "[完成] 报告已保存到: $OUTPUT_FILE"
```

执行：
```bash
chmod +x ./nginx_test.sh
./nginx_test.sh
```

### 测试三：iperf 测试


#### 1. 安装 iperf3（host 与 guest 均需执行）
```bash
sudo apt install -y iperf3
```

#### 2. host 启动 iperf3 服务端（绑定指定 IP）
```bash
iperf3 -s -B 192.168.101.2
```

#### 3. guest 运行压测脚本
将如下内容保存为 `iperf_test.sh`，并赋予可执行权限：
```bash
# 配置参数
HOST_IP="192.168.101.2"
PKT_SIZE=128       # 包的大小
THREADS=4          # 4 线程并发，防止单核 CPU 成为瓶颈
TIME=10            # 每档压测时长 10 秒

echo "=========================================================="
echo "  QEMU VirtIO 梯度压力测试 (UDP 小包注入)"
echo "  包大小: ${PKT_SIZE} Bytes | 并发线程: ${THREADS} | 时长: ${TIME}s"
echo "=========================================================="
printf "%-15s | %-15s | %-15s | %-15s\n" "目标注入带宽" "实际发送带宽" "实际接收(有效)带宽" "丢包率"
echo "----------------------------------------------------------"

# 定义带宽梯度 (单位: Gbps)
for bw in 1 2 5 8 10 15 20 25; do

  # 执行 iperf3 并捕获全部输出
  OUTPUT=$(iperf3 -c $HOST_IP -p 9999 -u -b ${bw}G -l $PKT_SIZE -P $THREADS -t $TIME)

  # 利用 awk 和 grep 提取多线程 [SUM] 汇总行的关键数据
  SEND_BW=$(echo "$OUTPUT" | grep "\[SUM\].*sender" | awk '{print $6 " " $7}')
  RECV_BW=$(echo "$OUTPUT" | grep "\[SUM\].*receiver" | awk '{print $6 " " $7}')
  LOSS_RATE=$(echo "$OUTPUT" | grep "\[SUM\].*receiver" | awk '{print $12}' | tr -d '()')

  # 错误处理：如果未能连通
  if [ -z "$RECV_BW" ]; then
    RECV_BW="测试失败/无数据"
    SEND_BW="-"
    LOSS_RATE="-"
  fi

  # 格式化打印结果
  printf "%-15s | %-15s | %-15s | %-15s\n" "${bw} Gbps" "$SEND_BW" "$RECV_BW" "$LOSS_RATE"

  # 冷却 3 秒，防止网卡队列积压影响下一轮
  sleep 3
done
echo "=========================================================="
```

执行：
```bash
chmod +x ./iperf_test.sh
./iperf_test.sh
```

```bash
#!/bin/bash
# VIO Sysbench 梯度测试汇总脚本
# 前提：Guest VM 已安装新版 sysbench 到 ~/opt/sysbench-master/bin

set -euo pipefail

# ---------- 配置 ----------
MYSQL_HOST="192.168.101.2"
MYSQL_PORT="3306"
MYSQL_USER="root"
MYSQL_PASS="123456"
MYSQL_DB="sbtest"
TABLE_SIZE=10000000
TABLES=64
TIME=120
THREADS_LIST=(1 10 32 64 128 256 512 840 1024)

# 使用手动编译的 sysbench
export PATH="$HOME/opt/sysbench-master/bin:$PATH"

# 基本检查
echo "=== 检查 sysbench 版本 ==="
sysbench --version

echo "=== 检查 MySQL 连通性 ==="
mysql -h"$MYSQL_HOST" -P"$MYSQL_PORT" -u"$MYSQL_USER" -p"$MYSQL_PASS" -e "SELECT 1;" || {
  echo "ERROR: 无法连接 MySQL，请检查网络、用户名密码"
  exit 1
}

# ---------- 场景一：oltp_read_only_point ----------
echo "========== 开始 read_only_point =========="
for i in "${THREADS_LIST[@]}"; do
  echo "===== read_only_point threads=$i ====="
  sysbench --thread-init-timeout=120 --db-driver=mysql \
    --mysql-host="$MYSQL_HOST" --mysql-port="$MYSQL_PORT" \
    --mysql-user="$MYSQL_USER" --mysql-password="$MYSQL_PASS" \
    --mysql-db="$MYSQL_DB" \
    --table_size="$TABLE_SIZE" --tables="$TABLES" \
    --time="$TIME" --threads="$i" --percentile=95 \
    --report-interval=1 --range_selects=0 --skip-trx=1 \
    oltp_read_only run
done | tee read_only_point.log

# ---------- 场景二：oltp_read_only ----------
echo "========== 开始 read_only =========="
for i in "${THREADS_LIST[@]}"; do
  echo "===== read_only threads=$i ====="
  sysbench --db-driver=mysql \
    --mysql-host="$MYSQL_HOST" --mysql-port="$MYSQL_PORT" \
    --mysql-user="$MYSQL_USER" --mysql-password="$MYSQL_PASS" \
    --mysql-db="$MYSQL_DB" \
    --table_size="$TABLE_SIZE" --tables="$TABLES" \
    --time="$TIME" --threads="$i" --percentile=95 \
    --report-interval=1 \
    oltp_read_only run
done | tee read_only.log

# ---------- 场景三：oltp_write_only ----------
echo "========== 开始 write_only =========="
for i in "${THREADS_LIST[@]}"; do
  echo "===== write_only threads=$i ====="
  sysbench --db-driver=mysql \
    --mysql-host="$MYSQL_HOST" --mysql-port="$MYSQL_PORT" \
    --mysql-user="$MYSQL_USER" --mysql-password="$MYSQL_PASS" \
    --mysql-db="$MYSQL_DB" \
    --table_size="$TABLE_SIZE" --tables="$TABLES" \
    --time="$TIME" --threads="$i" --percentile=95 \
    --report-interval=1 \
    oltp_write_only run
done | tee write_only.log

# ---------- 场景四：oltp_read_write ----------
echo "========== 开始 read_write =========="
for i in "${THREADS_LIST[@]}"; do
  echo "===== read_write threads=$i ====="
  sysbench --db-driver=mysql \
    --mysql-host="$MYSQL_HOST" --mysql-port="$MYSQL_PORT" \
    --mysql-user="$MYSQL_USER" --mysql-password="$MYSQL_PASS" \
    --mysql-db="$MYSQL_DB" \
    --table_size="$TABLE_SIZE" --tables="$TABLES" \
    --time="$TIME" --threads="$i" --percentile=95 \
    --report-interval=1 \
    oltp_read_write run
done | tee read_write.log

echo "========== 全部测试完成，日志文件："
ls -lh read_only_point.log read_only.log write_only.log read_write.log
```

执行：
```bash
chmod +x ./redis_test.sh
./redis_test.sh
```
